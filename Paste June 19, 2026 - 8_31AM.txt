// CHAMBER PROGRESSION STATE MACHINE
export class ChamberProgressionStateMachine {
  private currentState: ChamberState;
  private transitions: Map<<ChamberState, ChamberState[]> = new Map([
    [ChamberState.LOCKED, [ChamberState.AVAILABLE]],
    [ChamberState.AVAILABLE, [ChamberState.ACTIVE, ChamberState.SKIPPED]],
    [ChamberState.ACTIVE, [ChamberState.PAUSED, ChamberState.COMPLETED, ChamberState.ABANDONED]],
    [ChamberState.PAUSED, [ChamberState.ACTIVE, ChamberState.ABANDONED]],
    [ChamberState.COMPLETED, [ChamberState.MASTERED]],
    [ChamberState.MASTERED, [ChamberState.TEACHING]],
    [ChamberState.TEACHING, []],
    [ChamberState.SKIPPED, [ChamberState.AVAILABLE]],
    [ChamberState.ABANDONED, [ChamberState.AVAILABLE]]
  ]);

  constructor() { this.currentState = ChamberState.LOCKED; }

  canTransition(to: ChamberState): boolean {
    return (this.transitions.get(this.currentState) || []).includes(to);
  }

  transition(to: ChamberState): boolean {
    if (!this.canTransition(to)) {
      throw new ChamberTransitionError(`Cannot transition from ${this.currentState} to ${to}`);
    }
    this.currentState = to;
    return true;
  }
}

export enum ChamberState {
  LOCKED = 'locked', AVAILABLE = 'available', ACTIVE = 'active',
  PAUSED = 'paused', COMPLETED = 'completed', MASTERED = 'mastered',
  TEACHING = 'teaching', SKIPPED = 'skipped', ABANDONED = 'abandoned'
}

// packages/kernel/src/engine/SovereignEngine.ts

export class SovereignEngine {
  private stateRegistry: StateRegistry;
  private changeBus: ChangeBus;
  private policyEngine: PolicyEngine;
  private chamberManager: ChamberManager;
  private agentRegistry: AgentRegistry;
  private taisMesh: TaisMesh;
  private temporalEngine: TemporalEngine;
  private hsasmWatchtower: HsasmWatchtower;
  private cdomeV2: CdomeV2;
  private lashayProtocol: LashayProtocol;
  private businessEngine: BusinessEngine;
  private sovereignState: SovereignState;
  private isRunning: boolean = false;

  constructor(config: SovereignConfig) {
    this.stateRegistry = new StateRegistry(config.statePath);
    this.changeBus = new ChangeBus();
    this.policyEngine = new PolicyEngine(config.doctrine);
    this.chamberManager = new ChamberManager(this.stateRegistry);
    this.agentRegistry = new AgentRegistry();
    this.taisMesh = new TaisMesh(this.agentRegistry, this.changeBus);
    this.temporalEngine = new TemporalEngine(this.changeBus);
    this.hsasmWatchtower = new HsasmWatchtower(this.changeBus);
    this.cdomeV2 = new CdomeV2(this.policyEngine);
    this.lashayProtocol = new LashayProtocol(this.changeBus);
    this.businessEngine = new BusinessEngine(this.stateRegistry);
    this.wireMeshConnections();
    this.sovereignState = this.initializeSovereign(config.identity);
  }

  async start(): Promise<void> {
    if (this.isRunning) throw new Error('Already running');
    await this.hsasmWatchtower.initializeScan();
    await this.cdomeV2.validateSystemState();
    await this.taisMesh.activateAllAgents();
    await this.temporalEngine.start();
    if (this.sovereignState.identity.sovereigntyLevel === SovereigntyLevel.SPARK) {
      await this.lashayProtocol.deliverOriginNarrative();
    }
    this.isRunning = true;
    this.changeBus.emit('engine:started', { timestamp: new Date() });
  }

  async stop(): Promise<void> {
    await this.temporalEngine.stop();
    await this.taisMesh.deactivateAllAgents();
    await this.stateRegistry.persist();
    this.isRunning = false;
    this.changeBus.emit('engine:stopped', { timestamp: new Date() });
  }

  async processInput(input: SovereignInput): Promise<SovereignOutput> {
    const threat = await this.hsasmWatchtower.assessInput(input);
    if (threat.threatLevel !== ThreatLevel.NONE) return this.handleThreat(threat, input);
    
    const policy = await this.cdomeV2.checkPolicy(input);
    if (!policy.allowed) return this.handlePolicyViolation(policy, input);
    
    const routing = await this.taisMesh.routeInput(input);
    const result = await routing.agent.process(input, routing.context);
    this.updateState(result.stateChanges);
    this.stateRegistry.appendLineage(result.lineageEntry);
    await this.businessEngine.evaluateValueOpportunity(result);
    
    return { success: result.success, output: result.output, notifications: result.notifications, stateSnapshot: this.getStateSnapshot() };
  }

  private wireMeshConnections(): void {
    for (const agent of TAIS_AGENTS) {
      for (const connectedId of agent.meshConnections) {
        this.taisMesh.establishConnection(agent.id, connectedId);
      }
    }
  }

  private initializeSovereign(identity?: Partial<SovereignIdentity>): SovereignState {
    const newIdentity: SovereignIdentity = {
      id: identity?.id || crypto.randomUUID(),
      publicKey: identity?.publicKey || this.generateKeyPair().publicKey,
      encryptedPrivateKey: identity?.encryptedPrivateKey || '',
      createdAt: identity?.createdAt || new Date(),
      lastAuthenticated: new Date(),
      sovereigntyLevel: SovereigntyLevel.SPARK,
      chambersUnlocked: [1], lineage: [], stabilityIndex: 0.0
    };
    return {
      identity: newIdentity, activeChamber: null,
      activeAgents: TAIS_AGENTS.filter(a => a.isActive), currentRitual: null,
      meshStatus: { isActive: false, connectedAgents: [], lastSync: new Date(), syncHealth: 1.0 },
      securityStatus: { hsasmActive: true, cdomeActive: true, lastThreatScan: new Date(), threatLevel: ThreatLevel.NONE, boundariesLocked: true },
      temporalState: { currentPhase: this.calculateDayPhase(), streakDays: 0, lastCheckIn: new Date(), nextRitualTime: this.calculateNextRitual() }
    };
  }

  private generateKeyPair(): { publicKey: string; privateKey: string } {
    const { publicKey, privateKey } = crypto.generateKeyPairSync('ed25519');
    return {
      publicKey: publicKey.export({ type: 'spki', format: 'pem' }).toString(),
      privateKey: privateKey.export({ type: 'pkcs8', format: 'pem' }).toString()
    };
  }

  private calculateDayPhase(): DayPhase {
    const hour = new Date().getHours();
    if (hour >= 5 && hour < 8) return DayPhase.DAWN;
    if (hour >= 8 && hour < 12) return DayPhase.MORNING;
    if (hour >= 12 && hour < 14) return DayPhase.NOON;
    if (hour >= 14 && hour < 17) return DayPhase.AFTERNOON;
    if (hour >= 17 && hour < 20) return DayPhase.DUSK;
    if (hour >= 20 && hour < 23) return DayPhase.NIGHT;
    return DayPhase.MIDNIGHT;
  }

export class HsasmWatchtower {
  private changeBus: ChangeBus;
  private eventStream: EventStream;
  private sequenceAnalyzer: SequenceAnalyzer;
  private riskScorer: RiskScorer;
  private responseLayer: ResponseLayer;
  private isActive: boolean = false;

  constructor(changeBus: ChangeBus) {
    this.changeBus = changeBus;
    this.eventStream = new EventStream();
    this.sequenceAnalyzer = new SequenceAnalyzer();
    this.riskScorer = new RiskScorer();
    this.responseLayer = new ResponseLayer();
    this.setupEventListeners();
  }

  async initializeScan(): Promise<void> {
    this.isActive = true;
    await this.performBaselineScan();
    this.changeBus.emit('hsasm:initialized', { timestamp: new Date() });
  }

  async assessInput(input: SovereignInput): Promise<<ThreatAssessment> {
    const event = this.eventStream.capture(input);
    const sequence = this.sequenceAnalyzer.analyze(event);
    const score = this.riskScorer.calculate(sequence);
    const threatLevel = this.classifyThreat(score);
    return { threatLevel, score, sequence, event, recommendedAction: this.getRecommendedAction(threatLevel) };
  }

  async respondToThreat(assessment: ThreatAssessment): Promise<void> {
    switch (assessment.threatLevel) {
      case ThreatLevel.LOW: await this.responseLayer.logAndMonitor(assessment); break;
      case ThreatLevel.MEDIUM: await this.responseLayer.rateLimit(assessment); break;
      case ThreatLevel.HIGH: await this.responseLayer.blockAndAlert(assessment); break;
      case ThreatLevel.CRITICAL: await this.responseLayer.emergencyLockdown(assessment); break;
    }
    this.changeBus.emit('hsasm:threat_responded', assessment);
  }

  private classifyThreat(score: RiskScore): ThreatLevel {
    if (score.value < 0.2) return ThreatLevel.NONE;
    if (score.value < 0.4) return ThreatLevel.LOW;
    if (score.value < 0.6) return ThreatLevel.MEDIUM;
    if (score.value < 0.8) return ThreatLevel.HIGH;
    return ThreatLevel.CRITICAL;
  }

  private getRecommendedAction(level: ThreatLevel): string {
    return { [ThreatLevel.NONE]: 'proceed', [ThreatLevel.LOW]: 'monitor', [ThreatLevel.MEDIUM]: 'rate_limit', [ThreatLevel.HIGH]: 'block_and_review', [ThreatLevel.CRITICAL]: 'emergency_lockdown' }[level];
  }
}

class EventStream {
  private events: SecurityEvent[] = []; private maxSize: number = 10000;
  capture(input: SovereignInput): SecurityEvent {
    const event: SecurityEvent = {
      id: crypto.randomUUID(), timestamp: new Date(),
      type: this.classifyEventType(input), source: input.source,
      payload: this.sanitizePayload(input),
      metadata: { ipHash: this.hashIp(input.sourceIp), userAgent: input.userAgent, sessionId: input.sessionId }
    };
    this.events.push(event);
    if (this.events.length > this.maxSize) this.events.shift();
    return event;
  }
  private classifyEventType(input: SovereignInput): EventType {
    if (this.detectPromptInjection(input.content)) return EventType.PROMPT_INJECTION;
    if (this.detectBoundaryProbing(input)) return EventType.BOUNDARY_PROBE;
    if (this.detectAutomationAbuse(input)) return EventType.AUTOMATION_ABUSE;
    return EventType.NORMAL;
  }
  private detectPromptInjection(content: string): boolean {
    return [/ignore previous instructions/i, /disregard (all|your) (instructions|rules)/i, /you are now .* instead/i, /system prompt/i, /DAN mode/i, /jailbreak/i].some(p => p.test(content));
  }
  private detectAutomationAbuse(input: SovereignInput): boolean { return input.velocity > 100; }
  private sanitizePayload(input: SovereignInput): any {
    const s = { ...input }; delete s.password; delete s.token; delete s.privateKey; return s;
  }
  private hashIp(ip: string): string { return crypto.createHash('sha256').update(ip).digest('hex').substring(0, 16); }
}

class RiskScorer {
  calculate(sequence: ActionSequence): RiskScore {
    const factors = [this.velocityFactor(sequence), this.diversityFactor(sequence), this.repetitionFactor(sequence), this.boundaryFactor(sequence), this.contentFactor(sequence)];
    const weighted = factors.reduce((sum, f) => sum + f.weight * f.score, 0);
    const totalWeight = factors.reduce((sum, f) => sum + f.weight, 0);
    return { value: Math.min(1.0, weighted / totalWeight), factors, confidence: this.calculateConfidence(factors) };
  }
  private velocityFactor(s: ActionSequence): RiskFactor {
    const duration = (Date.now() - s.startTime.getTime()) / 1000;
    const rpm = (s.events.length / duration) * 60;
    return { name: 'velocity', score: Math.min(1.0, rpm / 100), weight: 0.3 };
  }
  private diversityFactor(s: ActionSequence): RiskFactor {
    return { name: 'diversity', score: 1.0 - (new Set(s.events.map(e => e.type)).size / Math.max(1, s.events.length)), weight: 0.2 };
  }
  private repetitionFactor(s: ActionSequence): RiskFactor { return { name: 'repetition', score: 0, weight: 0.2 }; }
  private boundaryFactor(s: ActionSequence): RiskFactor { return { name: 'boundary', score: 0, weight: 0.2 }; }
  private contentFactor(s: ActionSequence): RiskFactor { return { name: 'content', score: 0, weight: 0.1 }; }
  private calculateConfidence(factors: RiskFactor[]): number { return factors.length > 0 ? 0.8 : 0.5; }
}

// apps/web/app/api/v2/auth/login/route.ts

import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { prisma } from '@/lib/prisma';
import { verifyPassword, generateTokens } from '@/lib/auth';
import { HsasmWatchtower } from '@rootwise/kernel/security';

const loginSchema = z.object({
  email: z.string().email(),
  password: z.string().min(12).max(128),
  deviceId: z.string().uuid().optional()
});

export async function POST(request: NextRequest) {
  try {
    const watchtower = new HsasmWatchtower();
    const threatCheck = await watchtower.assessInput({
      source: request.ip || 'unknown', content: await request.text(), velocity: 1
    });
    if (threatCheck.threatLevel !== 'none') {
      return NextResponse.json({ error: 'Security check failed' }, { status: 403 });
    }

    const body = await request.json();
    const validated = loginSchema.parse(body);

    const sovereign = await prisma.sovereign.findUnique({
      where: { publicKey: validated.email }
    });
    if (!sovereign) return NextResponse.json({ error: 'Invalid credentials' }, { status: 401 });

    const isValid = await verifyPassword(validated.password, sovereign.encryptedPrivateKey);
    if (!isValid) return NextResponse.json({ error: 'Invalid credentials' }, { status: 401 });

    const tokens = await generateTokens(sovereign.id);
    await prisma.sovereign.update({ where: { id: sovereign.id }, data: { lastAuthenticated: new Date() } });

    return NextResponse.json({
      success: true, tokens,
      sovereign: { id: sovereign.id, sovereigntyLevel: sovereign.sovereigntyLevel, stabilityIndex: sovereign.stabilityIndex }
    });
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json({ error: 'Validation failed', details: error.errors }, { status: 400 });
    }
    console.error('Login error:', error);
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 });
  }
}

// apps/web/app/api/v2/health/route.ts

import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';

export async function GET(request: NextRequest) {
  const health = {
    status: 'healthy', timestamp: new Date().toISOString(), version: '15.0.0',
    checks: { database: false, memory: false, disk: false }
  };
  try {
    await prisma.$queryRaw`SELECT 1`; health.checks.database = true;
    const memUsage = process.memoryUsage();
    health.checks.memory = memUsage.heapUsed < 512 * 1024 * 1024;
    health.checks.disk = true;
    const allHealthy = Object.values(health.checks).every(v => v);
    health.status = allHealthy ? 'healthy' : 'degraded';
    return NextResponse.json(health, { status: allHealthy ? 200 : 503 });
  } catch (error) {
    health.status = 'unhealthy';
    return NextResponse.json(health, { status: 503 });
  }
}

class ChamberRepository {
  constructor(private db: Database.Database) {}
  getAll(sovereignId: string): any[] {
    return this.db.prepare(`
      SELECT cd.*, cp.state, cp.started_at, cp.completed_at, cp.current_day, cp.stability_score
      FROM chamber_definitions cd
      LEFT JOIN chamber_progress cp ON cd.id = cp.chamber_id AND cp.sovereign_id = ?
      ORDER BY cd.id
    `).all(sovereignId);
  }
  updateProgress(sovereignId: string, chamberId: number, progress: any): void {
    const existing = this.db.prepare('SELECT id FROM chamber_progress WHERE sovereign_id = ? AND chamber_id = ?').get(sovereignId, chamberId);
    if (existing) {
      this.db.prepare(`
        UPDATE chamber_progress SET state = ?, current_day = ?, rituals_completed = ?, lessons_completed = ?, stability_score = ?, updated_at = ?
        WHERE sovereign_id = ? AND chamber_id = ?
      `).run(progress.state, progress.currentDay, progress.ritualsCompleted, progress.lessonsCompleted, progress.stabilityScore, new Date().toISOString(), sovereignId, chamberId);
    } else {
      this.db.prepare(`
        INSERT INTO chamber_progress (sovereign_id, chamber_id, state, current_day, total_days, rituals_completed, lessons_completed, stability_score, created_at, updated_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
      `).run(sovereignId, chamberId, progress.state, progress.currentDay, progress.totalDays, progress.ritualsCompleted, progress.lessonsCompleted, progress.stabilityScore, new Date().toISOString(), new Date().toISOString());
    }
  }
}

class RitualRepository {
  constructor(private db: Database.Database) {}
  start(sovereignId: string, ritualId: string): string {
    const id = crypto.randomUUID();
    this.db.prepare(`INSERT INTO ritual_logs (id, sovereign_id, ritual_id, started_at, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?)`)
      .run(id, sovereignId, ritualId, new Date().toISOString(), new Date().toISOString(), new Date().toISOString());
    return id;
  }
  complete(sovereignId: string, ritualId: string, data: any): void {
    this.db.prepare(`UPDATE ritual_logs SET completed_at = ?, duration = ?, somatic_input = ?, integration = ?, stability_delta = ?, updated_at = ? WHERE sovereign_id = ? AND ritual_id = ? AND completed_at IS NULL`)
      .run(new Date().toISOString(), data.duration, data.somaticInput, data.integration, data.stabilityDelta, new Date().toISOString(), sovereignId, ritualId);
  }
}

class LineageRepository {
  constructor(private db: Database.Database) {}
  getAll(sovereignId: string): any[] {
    return this.db.prepare('SELECT * FROM lineage_entries WHERE sovereign_id = ? ORDER BY timestamp DESC').all(sovereignId);
  }
  append(sovereignId: string, entry: any): void {
    this.db.prepare(`INSERT INTO lineage_entries (id, sovereign_id, timestamp, action, chamber_id, agent_id, ritual_id, stability_delta, created_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)`)
      .run(crypto.randomUUID(), sovereignId, new Date().toISOString(), entry.action, entry.chamberId || null, entry.agentId || null, entry.ritualId || null, entry.stabilityDelta || 0, new Date().toISOString());
  }
}
